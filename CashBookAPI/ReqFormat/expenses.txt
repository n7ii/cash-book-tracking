## get Expense: 

*GET /api/expenses?startDate=YYYY-MM-DD&endDate=YYYY-MM-DD&page=1&search=fuel

final headers = {
  'Authorization': 'Bearer $token',   // Needed to prove who the user is
};





## get single expense: 

*GET /api/expenses/:id

final headers = {
  'Content-Type': 'application/json', 
};





## Add new Expense: 

POST /api/expenses

final headers = {
  'Content-Type': 'application/json', 
  'Authorization': 'Bearer $token', 
};

final body = jsonEncode({
    {
    "expense_type": "Fuel",
    "amount": 30000,
    "notes": "Filled up the tank for market visits.",
    "photo_url": null,
    "category": "Transportation",         
    "payment_method": "Cash",             
    "market_id": 2     
}
    });





## update Expense: 

*PUT /api/expenses/:id

final headers = {
  'Content-Type': 'application/json', // Needed if sending a JSON body
  'Authorization': 'Bearer $token',   // Needed to prove who the user is
};

final body = jsonEncode({
    {
    "expense_type": "Fuel",
    "amount": 35000,
    "notes": "Corrected amount from receipt.",
    "category": "Transportation",         // NEW
    "payment_method": "Cash",             // NEW
    "market_id": 2,                       // NEW
    "edit_reason": "Typo in the original amount."
}
    });
